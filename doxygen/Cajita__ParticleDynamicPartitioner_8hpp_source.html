79  void compute( Kokkos::View<int***, memory_space>& workload ) override
80  {
81  Kokkos::Array<CellUnit, num_space_dim> lower_corner;
82  for ( std::size_t d = 0; d < num_space_dim; ++d )
83  {
84  lower_corner[d] = global_lower_corner[d];
85  }
86 
87  auto dx_copy = dx;
88  auto cell_bits_per_tile_dim_copy = cell_bits_per_tile_dim;
89  auto view_copy = view;
90  Kokkos::parallel_for(
91  "compute_local_workload_parpos",
92  Kokkos::RangePolicy<execution_space>( 0, particle_num ),
93  KOKKOS_LAMBDA( const int i ) {
94  int ti = static_cast<int>(
95  ( view_copy( i, 0 ) - lower_corner[0] ) / dx_copy -
96  0.5 ) >>
97  cell_bits_per_tile_dim_copy;
98  int tj = static_cast<int>(
99  ( view_copy( i, 1 ) - lower_corner[1] ) / dx_copy -
100  0.5 ) >>
101  cell_bits_per_tile_dim_copy;
102  int tz = static_cast<int>(
103  ( view_copy( i, 2 ) - lower_corner[2] ) / dx_copy -
104  0.5 ) >>
105  cell_bits_per_tile_dim_copy;
106  Kokkos::atomic_increment( &workload( ti + 1, tj + 1, tz + 1 ) );
107  } );
108  Kokkos::fence();
109  // Wait for other ranks' workload to be ready
110  MPI_Barrier( comm );
111  }
112 };
