61  void compute( Kokkos::View<int***, memory_space>& workload ) override
62  {
63  const SparseMapType& sparse_map_copy = sparseMap;
64  Kokkos::parallel_for(
65  "compute_local_workload_sparsmap",
66  Kokkos::RangePolicy<execution_space>( 0, sparseMap.capacity() ),
67  KOKKOS_LAMBDA( uint32_t i ) {
68  if ( sparse_map_copy.valid_at( i ) )
69  {
70  auto key = sparse_map_copy.key_at( i );
71  int ti, tj, tk;
72  sparse_map_copy.key2ijk( key, ti, tj, tk );
73  Kokkos::atomic_increment(
74  &workload( ti + 1, tj + 1, tk + 1 ) );
75  }
76  } );
77  Kokkos::fence();
78  // Wait for other ranks' workload to be ready
79  MPI_Barrier( comm );
80  }
81 };
